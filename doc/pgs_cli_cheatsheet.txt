# change_mgmt — SDLC lifecycle: ONE CLI, FOUR verbs  (author -> construct -> validate -> promote)
# Run from ~/pgs/pgs_change_mgmt with the venv active. PGS_WORKSPACE must be declared (read-only pi grounding).
# `pgs_change <verb>` after the next editable reinstall; `python -m pgs_change_mgmt.engine.cli <verb>` works today.
# User thinks in PHASES; stages S1..S9 + compiler mechanics (build/admit/persist) stay internal (--stage = debug only).
# `pgs_construct` is a deprecated alias for `construct`.

export PGS_WORKSPACE=~/pgs/pgs_workspace

# 1) AUTHOR  (Design: S1..S7 -> CR-IR).  Automated worker by default; --guided = interactive export/import.
#    --worker/--model REQUIRED (a conscious choice).  stage order: 1 2 3 4 5 6 6b 7
python -m pgs_change_mgmt.engine.cli author --worker qwen --model qwen3.5:latest              # full S1..S7
python -m pgs_change_mgmt.engine.cli author --worker qwen --model qwen3.5:latest --stage 6b   # one stage (debug)
#    -- guided authoring (human / Claude-Code is the worker): one governed Stage Package per stage --
python -m pgs_change_mgmt.engine.cli author --guided --seed blockchain_chain --stage 1 --export
#      -> paste the model's full reply into <dossier>/_packages/stage_<N>/response.md  (ground via pi as you go)
python -m pgs_change_mgmt.engine.cli author --guided --seed blockchain_chain --stage 1 --import --model-label claude-code

# 2) CONSTRUCT  (S8 + Admission: CR-IR -> ADMITTED_UNVALIDATED).  build->admit as ONE verb (admit is internal).
#    writes finale .md to --out (default constructed/, the input to promote); --persist writes a full test-snapshot.
python -m pgs_change_mgmt.engine.cli construct --projection change_mgmt/dossiers/blockchain/chain/cr_ir --domain blockchain --subdomain chain
python -m pgs_change_mgmt.engine.cli construct --projection change_mgmt/dossiers/blockchain/chain/cr_ir --domain blockchain --subdomain chain --persist /abs/test_ws
#    -> Construction Status: CONSTRUCTION COMPLETE | Admission: PASS | Snapshot: ADMITTED_UNVALIDATED  (exit 0; 2 = INVALID)

# 3) VALIDATE  (Execution Validation -> EXECUTION_VALIDATED).  Runs the CR's declared acceptance scenario.
python -m pgs_change_mgmt.engine.cli validate --dossier blockchain/chain          # -> Validation Status: PASS

# 4) PROMOTE  (S9 -> PROMOTED).  KISS: gated on PASS evidence + --confirm; copy finale .md to the domain SOURCE
#    registry, then the compiler is the gate (compile -> build -> pi validate --strict); rollback-all-on-red.
#    --registry-root REQUIRED (the protocol SOURCE, never protocol_snapshot/).  Rebuilds the whole workspace snapshot.
python -m pgs_change_mgmt.engine.cli promote --dossier blockchain/chain --registry-root ~/pgs/pgs_blockchain/pgs_blockchain/registry            # gate preview (no mutation)
python -m pgs_change_mgmt.engine.cli promote --dossier blockchain/chain --registry-root ~/pgs/pgs_blockchain/pgs_blockchain/registry --confirm  # perform S9

# -- workspace shell wrapper (sets PGS_WORKSPACE + venv): ~/pgs/pgs_workspace/scripts/run_change_mgmt.sh <verb> ... --
#    (older run_change_mgmt_dossier.sh / _interactive.sh still work as deprecated forwarders to `author` / `author --guided`)

# -- cr_ir/ = the governed per-stage handoffs = the Construction Projection S8 consumes (TRACKED; source of truth) --
#   layout: change_mgmt/dossiers/<domain>/<subdomain>/cr_ir/<stage>.json
# REPAIR ONLY (NOT a pipeline stage): regenerate cr_ir/*.json from the locked markdown baseline.
#   Migration can be lossy — never run against a good golden unless you intend to overwrite it.
python -m pgs_change_mgmt.engine._migrate_cr_ir_from_baseline --dossier change_mgmt/dossiers/blockchain/chain --stages 5 6b 7

# -- internal / advanced (construct dispatches into these; use directly only for isolated debugging) --
#   raw Construction Compiler subcommands: build | validate | explain | graph | admit (admit = zero-pollution gate, no repo writes)
python -m pgs_change_mgmt.engine.construction_cli explain --projection change_mgmt/dossiers/blockchain/chain/cr_ir --domain blockchain --subdomain chain   # per-pass status + violation detail (GAP_DOSSIER etc.)
python -m pgs_change_mgmt.engine.construction_cli graph   --projection change_mgmt/dossiers/blockchain/chain/cr_ir --domain blockchain --subdomain chain   # node/edge dump
#   worker-driven S8->S9 transcription variant (ollama + bridge promote + S9 Construction Record + semantic-preservation gate):
python -m pgs_change_mgmt.engine.construct_chain --registry-root ~/pgs/pgs_blockchain/pgs_blockchain/registry --models qwen3:14b --kinds CC

# -- selftests (hermetic; the change_mgmt engine regression suite) --
bash ~/pgs/pgs_change_mgmt/scripts/verify_change_mgmt_engine.sh

# pi — Protocol Inspection (read-only queries over the compiled snapshot set)
# Setup: workspace must be declared (no cwd guessing). One-time:

export PGS_WORKSPACE=~/pgs/pgs_workspace

# Every command also accepts --workspace <abs path> and --json (stable schema for agents/CI/jq).
# Full FQDNs required in one-shot mode; bare codes work only inside the shell with a declared scope.


# -- interactive shell (tab completion over objects, verbs, FQDNs) --

pi                                      # no args opens the shell
#   pi> use blockchain                  # declare scope -> prompt becomes pi:blockchain>
#   pi:blockchain> artifact refs CC_GENERATE_TX_ID_V0    # bare codes expand within scope
#   pi:blockchain> behavior_logic show WF_MINT_V0
#   pi:blockchain> use                  # clear scope
#   pi> exit


# -- artifact (kind-agnostic, the daily drivers) --

pi artifact show      blockchain::CC_GENERATE_TX_ID_V0     # identity, governance, ref/dep counts
pi artifact refs      blockchain::CC_GENERATE_TX_ID_V0     # who references this (consumers)
pi artifact refs      blockchain::CC_GENERATE_TX_ID_V0 --transitive
pi artifact deps      blockchain::WF_CREATE_WALLET_V0 --transitive
pi artifact lineage   capability_side_effects::CS_APPENDONLY_JSONL_V0   # ancestor + descendant trees
pi artifact owner     blockchain::WF_CREATE_WALLET_V0      # domain / subdomain / STRUCTURE
pi artifact source    blockchain::WF_MINT_V0               # authoring Markdown from PPS snapshot
pi artifact lifecycle blockchain::WF_CREATE_WALLET_V0      # declared status (ACTIVE/RETIRED/UNKNOWN)
pi artifact list --kind WF --domain blockchain


# -- kind objects (list/show sugar + kind-specific verbs) --

pi wf list --domain ai_governance
pi wf lineage  blockchain::WF_CREATE_WALLET_V0             # execution tree from evidence routing
pi wf outcomes blockchain::WF_CREATE_WALLET_V0             # declared terminal states
pi cc outcomes blockchain::CC_GENERATE_TX_ID_V0            # enumerated outcome set
pi cc binding  blockchain::CC_DERIVE_WALLET_KEYS_V0        # CT/CS bindings + mapping RBs
pi ct impl     capability_transforms::CT_PURE_GENERATE_ID_V0   # RBs mapping this CT
pi cs surface                                              # the enumerated side-effect surface
pi rb resolve  capability_side_effects::CS_SEND_EMAIL_V0   # which RBs bind this CS
pi ev list --domain blockchain
pi ac show blockchain::AC_ENDUSER_V0
pi in show blockchain::IN_WALLET_CREATED_V0


# -- topology (change-management workhorse; impact output is dossier evidence) --

pi topology wf     blockchain::WF_MINT_V0                  # full reachability graph of a WF
pi topology cc     blockchain::CC_GENERATE_TX_ID_V0        # every WF position with routing context
pi topology impact capability_transforms::CT_PURE_GENERATE_ID_V0       # transitive consumer closure
pi topology impact capability_transforms::CT_PURE_GENERATE_ID_V0 --json   # paste into dossier
pi topology path   blockchain::WF_MINT_V0 capability_side_effects::CS_APPENDONLY_JSONL_V0


# -- behavior logic (reads *.graph.json, never the PNGs) --

pi behavior_logic list
pi behavior_logic show   blockchain::WF_VERIFY_ACTOR_V0    # terminal tree incl. EXIT routing
pi behavior_logic render blockchain::WF_MINT_V0 --mermaid  # paste into Markdown
pi behavior_logic render blockchain::WF_MINT_V0 --dot      # pipe to graphviz
pi behavior_logic open   blockchain::WF_CREATE_WALLET_V0   # open compiler-materialized PNG


# -- store (entity-store ownership; from artifact_index/stores.json) --

pi store list
pi store show      ACTOR                                   # bare name OK when unique; else domain::STORE
pi store show      GOVERNANCE_AUDIT                        # shows ALL declarations (multi-path case)
pi store consumers ACTOR                                   # CCs that touch this store


# -- vocab / pps (address space + authoring surface) --

pi vocab search  WALLET
pi vocab resolve blockchain::WF_MINT_V0                    # FQDN -> structure, paths, addresses
pi vocab stats
pi pps stats
pi pps list --kind WF
pi pps show blockchain::CC_GENERATE_TX_ID_V0


# -- snapshot / validation / CI --

pi stats                                # workspace-wide one-screen summary
pi snapshot status                      # snapshot_status.json: valid? when? hash?
pi snapshot summary                     # counts by kind and domain
pi snapshot topology                    # domain -> subdomain -> WF map
pi snapshot stats                       # graph metrics (orphans = ASSERTs + TIs, expected)
pi snapshot validate                    # conformance state (pure query, always exit 0)
pi snapshot violations                  # failing cases named; exit 0
pi snapshot violations --strict         # exit 1 if any violations (CI gate)
pi validate --strict                    # one-shot CI gate: VALID + zero violations or exit 1

pi trace list
